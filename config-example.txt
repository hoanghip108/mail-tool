# Cấu hình Email SMTP
# Copy nội dung này vào file .env và điền thông tin của bạn

# Gmail: smtp.gmail.com (port 587)
# Outlook: smtp-mail.outlook.com (port 587)
# Yahoo: smtp.mail.yahoo.com (port 587)
SMTP_HOST=smtp.gmail.com
SMTP_PORT=587
SMTP_SECURE=false

# Email và mật khẩu của bạn
# Lưu ý: Với Gmail, bạn cần tạo App Password tại https://myaccount.google.com/apppasswords
SMTP_USER=your-email@gmail.com
SMTP_PASS=your-app-password

# Thông tin người gửi
FROM_NAME=Storydesk
FROM_EMAIL=your-email@gmail.com

# Chủ đề email
EMAIL_SUBJECT=Xác nhận đơn hàng của bạn
